function calc() { const H = parseFloat(document.getElementById("high").value); const L = parseFloat(document.getElementById("low").value); const C = parseFloat(document.getElementById("close").value); if (isNaN(H) || isNaN(L) || isNaN(C)) { alert("กรอกตัวเลข High / Low / Close ให้ครบ และห้ามมี ,"); return; } const P = (H + L + C) / 3; const R1 = 2*P - L; const S1 = 2*P - H; const R2 = P + (H - L); const S2 = P - (H - L); const R3 = H + 2*(P - L); const S3 = L - 2*(H - P); const levels = [ ["R3", R3], ["R2", R2], ["R1", R1], ["P", P], ["S1", S1], ["S2", S2], ["S3", S3] ]; document.getElementById("out").innerHTML = levels.map(x => `${x[0]}${x[1].toFixed(2)}` ).join(""); }
